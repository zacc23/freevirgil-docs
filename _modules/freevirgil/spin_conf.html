[docs]    def dec_conf(self, dec):
        """
        Convert decimal to binary (-1, 1)
        spin configuration
        
        Parameters
        ----------
        dec : int
            Decimal number to convert to
            binary (0, 1) -> (-1, 1)

        Returns
        -------
        conf : int list
            Spin configuration list (-1, 1)
        """
        conf = [int(i) for i in bin(dec)[2:]]
        
        for i in range(0, len(conf)):
            if conf[i] == 0:
                conf[i] = -1
        
        while len(conf) < self.N:
            conf = [-1] + conf
            
        return conf
